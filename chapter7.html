Q4.Write a program to find whether a given number is prime or not.

num = int(input("Enter the number "))
prime = True

for i in range(2,num):
	if(num%i==0):
		prime = False
	break
	
if prime == True:
	print("This is a prime number")
else:
	print("This is not a prime number ")

Q5.Write a program to find the sum of first n natural numbers using while loop.

num = int(input("Enter the number "))
sum = 0
if(num<=0):
	print("Enter a positive number")
else:
	while num>0:
		
		sum = sum + num
		num-=1
	print(sum)

Q6.Write a program to calculate the factorial of a given number using for loop

num = int(input("Enter a number "))

fac = 1
for i in range(1,num+1):
	fac = fac*i 
print(f"{num}! = {fac}")

Q7.Write a python program to print the following star pattern
*
**
***

num = int(input("Enter number"))
for i in range(1,num+1):
	print("*"*i)

Q8.Write a python program to print the following star pattern.
  *  
 *** 
*****

pat = input("Which pattern you want to print")
n = int(input("how many lines"))

for i in range(n):
	print(" "*(n-i-1), end="") #end="" prevent printing new line
	print(pat*(2*i+1),end = "")
	print(" "*(n-i-1))

Q9.Write a python program to print the following star pattern.
* * *
*   *
* * *

n = int(input("Enter the row and col numbers "))
      #for this question n = 3
