Q1.Write a program to find greatest of three nubers.

def greatest(a,b,c):
	list = [a,b,c]
	list.sort() #update list in increasing order
	return list[-1] # -1 index means last number

num = greatest(100,31,11)
print(num)

Q4.Write a recursive function to calculate the sum of first n natural numbers.

def sum_nat(n):
	if n==0:
		return  0
	else:
		return n+sum_nat(n-1)
result = sum_nat(5)
print(result)

Q7.Write a python function to remove a given word from a string and strip it at the same time.

def re_strip(word):
	string = "    harry is a good boy    "
	string = string.replace(word,"")
	string = string.strip()
	return string
	
a = re_strip("good")

print(a)
