Q2.Write a class calculator capable of finding Square, Sq.Root and Cube of a given number.

class calculator:
	def __init__(self,number):
		self.num = number
	
	def square(self):
		print(f"Squre of {self.num} is {self.num**2}")
	def cube(self):
		print(f"Cube of {self.num} is {self.num**3}")
	def squareRoot(self):
		print(f"Square root of {self.num} is {self.num**0.5}")

inpNum = int(input("Enter number "))
a = calculator(inpNum)
a.square()
a.cube()
a.squareRoot()

Q4.Add a static method in problem 2 to greet the user with Hello.

class calculator:
	def __init__(self,number):
		self.num = number
	
	def square(self):
		print(f"Squre of {self.num} is {self.num**2}")
	def cube(self):
		print(f"Cube of {self.num} is {self.num**3}")
	def squareRoot(self):
		print(f"Square root of {self.num} is {self.num**0.5}")
	@staticmethod #staticmethod doesnot take self in argument
	def greet():
		print("Hello \n This is the best calculator by python")

inpNum = int(input("Enter number "))
a = calculator(inpNum)
a.square()
a.cube()
a.squareRoot()
a.greet()
